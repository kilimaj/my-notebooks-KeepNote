# Lists - Have brackets [ ]

movies = ["Harry Porter", "The Hangover", "Wildflower", "The Jocker"]

print(movies[0]) # prints the first item
print(movies[0:3]) # prints first item to third item i.e. item 0, 1, 2 item 3 is not included
print(movies[-1]) # last item

print(len(movies))
movies.append("SAW")
print(movies)

movies.pop() # removes last item
print(movies)

movies.pop(1) # removes second item
print(movies)
