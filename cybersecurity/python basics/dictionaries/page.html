#!/bin/python3

# Dictionaries - key:value pairs {}

drinks = {"White Russian": 7, "Old Fashion": 10, "Lemon Drop": 8} #drink is key and price is value
print(drinks)

employees = {"Finance": ["Bob", "Linda", "Tina"], "IT": ["Gene", "Louise", "Teddy"], "HR": ["Jimmy", "John"]}

print(employees)

employees['Legal'] = ["Mr. John"] #add a new key :value pair

print(employees)

employees.update({"Sales": ["John", "Noella"]}) #add a new key :value pair

print(employees)

drinks['White Russian'] = 8

print(drinks)

print(drinks.get("White Russian"))
